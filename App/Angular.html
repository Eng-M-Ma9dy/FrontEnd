                            <h1>Condition False</h1>
                        </div>
                    </div>

                    <!-------------------------------------------------------------------------------------------->

                    <!-- Notes -->
                    <!-- Structure Directives -> *ngIF      : The HTML File Not Included Of This Element -->
                    <!-- Attribute Directives -> [ngClass]  : The HTML File Included This Element But The Element Take class = "d-none"  -->

                    <!-------------------------------------------------------------------------------------------->

                    <!-- If I Want Apply The Same Condition on More Than One Element, I Have 2 Solution -->

                    <!-- 1. Set All Elements In Div, and Set The Condition The Div -->
                    <!------But In This Case I Create Div Unnecessarily ------------->

                    <!-- 2. Set All Elements In <ng-template>, and Set The Condition The Template ----------------------->
                    <!------In This Case I Apply Condition on All Element Without Create New Element--------------------->
                    <!------And I Can Create The Template and Set Template Name, For Use with *ngIf in Condition Else---->

                    <div class="w-75 mx-auto text-white border border-5 border-light my-5 py-5">
                        <h4 class="py-3"> Template [ngIf]</h4>
                        <ng-template [ngIf]=" testAge < 30 ">
                            <h1>{{testName}}</h1>
                            <h1>{{testAge}}</h1>
                        </ng-template>
                    </div>

                    <!-- Else Condition------------------------------------------------------------------------------------------>
                    <!-- In This Case If Condition Else Display Another Template -->

                    <div class="w-75 mx-auto text-white border border-5 border-light my-5 py-5">
                        <h4 class="py-5">(*ngIf) Else : Display Template</h4>

                        <ng-template *ngIf=" testAge > 30 ; else megaTemplate ">
                            <h1>{{testName}}</h1>
                            <h1>{{testAge}}</h1>
                        </ng-template>

                        <ng-template #megaTemplate>
                            <div class="w-75 mx-auto text-white py-2">

                        </ng-template>

    import { Component, OnDestroy, OnInit } from '@angular/core';
    import { MoviesApiService } from '../movies-api.service';
    
    // --------------------------------------------------------
    // Call API :
    // -----------
     
    // - For call API go to app.module.ts and make import { HttpClientModule } from "@angular/common/http"; and imports:[HttpClientModule]
    // - Create Service (MoviesApiService) For Call API and Make Export and Import in This File
    // - Create Function For Call API And Call it in ngOnInit() 
    
    @Component({
      selector: 'app-movies',
      templateUrl: './movies.component.html',
      styleUrls: ['./movies.component.css']
    })
    export class MoviesComponent implements OnInit, OnDestroy {
    
      allMovies:any = [];
    
      isLoading: Boolean = true;
    
      apiCall:any
    
      constructor(private _MoviesService:MoviesApiService) { }
    
      ngOnInit(): void {
    
        this.getAllData()
    
      }
    
      getAllData(){
    
        this.apiCall = this._MoviesService.getMovies().subscribe(
    
          // .subscribe() for call API (No Call Without subscribe())
    
          (res)=>{
    
            this.allMovies = res.results
    
            console.log(this.allMovies);
    
            this.isLoading = false
            
          },
    
          (err)=>{
            
            this.isLoading = false,
    
            console.log(err)
          
          },
    
          ()=>console.log("Completed")
          
        )
    
      }
    
      ngOnDestroy(): void {
        this.apiCall.unsubscribe()
      }
    
    }
