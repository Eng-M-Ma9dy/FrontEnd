    .parent{

        //-----------------------------------------------------------------
        // Do For Loop
        //---------------
        // From 1  : First Ele 1
        // To   10 : Last  Ele 9
    
        @for $i from 1 to 10 {
    
            .child_#{$i}{
    
                // For Use Color Name By Hexadecimal Without Convert
    
                // $i/10 : For Change Value Of Alpha Every Round
    
                background-color: rgba($color: #08c, $alpha: $i / 10);
                width: 50px + ( 30px * $i );
                text-align: center;
                color: black;
                padding: 10px;
                border: none;
    
            }
    
        }
    
        //-----------------------------------------------------------------
        // For Make Array 
        //----------------
    
        $children: child_1 , child_3 , child_5 , child_7 , child_9 ;
    
        // For Loop on Array
    
        // $class The Name For Every Ele in Array
    
        // $children name of Array i want do loop
    
        @each $class in $children {
    
            .#{$class}{
                border-top: solid black 5px;
            }
    
        }
    
        //-----------------------------------------------------------------
        // IF Else
        //----------
    
        $left:left;
    
        $right:right;
        
        .left{
    
            width: 300px;
            height: 150px;
            overflow: hidden;
            background-color: teal;
    
            float: $left;
    
            padding-#{$left}: 40px ;
            margin-#{$left}: 40px ;
        
        }
        
        .right{
    
            width: 300px;
            height: 150px;
            overflow: hidden;
            background-color: tomato;
    
            float: $right;
    
            padding-#{$right}: 40px ;
            margin-#{$right}: 40px ;
        
        
        }
    
        $dir:rtl;
        
        @if $dir == ltr {
        
            $left:left;
            $right:right;
        
        }@else{
        
            $left:right;
            $right:left;
        }
    
    }

    @function convertPxToRem($value){
    
        $res : $value / 16;
    
        @return #{$res}rem;
    
    }

        font-size: convertPxToRem(70);
    }

    @mixin changeArrowDirection ($direction , $color , $size) {

        width: 0;
        height: 0;
        border: solid transparent #{$size}px;
    
        
    
        @if $direction == left {
    
            border-right-color: $color;
    
        }@else if $direction == right {
    
            border-left-color: $color;
    
        }@else if $direction == top {
    
            border-bottom-color: $color;
    
        }@else if $direction == bottom {
    
            border-top-color: $color;
    
        }
    
    }
    
    
    #brd{
    
        @include changeArrowDirection( top , white ,150)
    
    }
